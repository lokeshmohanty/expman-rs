17pub async fn list_experiments(State(state): State<AppState>) -> impl IntoResponse {
18    match storage::list_experiments(&state.base_dir) {
19        Ok(names) => {
20            let mut result = vec![];
21            for name in &names {
22                let exp_dir = exp_dir(&state.base_dir, name);
23                let runs = storage::list_runs(&exp_dir).unwrap_or_else(|_| vec![]);
24                let meta = storage::load_experiment_metadata(&exp_dir).unwrap_or_default();
25                result.push(serde_json::json!({
26                    "id": name,
27                    "display_name": meta.display_name.unwrap_or_else(|| name.to_string()),
28                    "description": meta.description,
29                    "tags": meta.tags,
30                    "runs_count": runs.len(),
31                }));
32            }
33            Json(result).into_response()
34        }
35        Err(e) => (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()).into_response(),
36    }
37}
38
39pub async fn get_experiment_metadata(
40    State(state): State<AppState>,
41    Path(exp): Path<String>,
42) -> impl IntoResponse {
43    let dir = exp_dir(&state.base_dir, &exp);
44    match storage::load_experiment_metadata(&dir) {
45        Ok(meta) => Json::<crate::core::models::ExperimentMetadata>(meta).into_response(),
46        Err(e) => (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()).into_response(),
47    }
48}

50#[derive(Deserialize)]
51pub struct MetadataUpdate {
52    pub display_name: Option<String>,
53    pub description: Option<String>,
54    pub tags: Option<Vec<String>>,
55}
56
57pub async fn update_experiment_metadata(
58    State(state): State<AppState>,
59    Path(exp): Path<String>,
60    Json(update): Json<MetadataUpdate>,
61) -> impl IntoResponse {
62    let dir = exp_dir(&state.base_dir, &exp);
63    let mut meta = storage::load_experiment_metadata(&dir).unwrap_or_default();
64    if let Some(dn) = update.display_name {
65        meta.display_name = Some(dn);
66    }
67    if let Some(desc) = update.description {
68        meta.description = Some(desc);
69    }
70    if let Some(tags) = update.tags {
71        meta.tags = tags;
72    }
73    match storage::save_experiment_metadata(&dir, &meta) {
74        Ok(_) => Json(meta).into_response(),
75        Err(e) => (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()).into_response(),
76    }
77}
