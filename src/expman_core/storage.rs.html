27pub fn list_experiments(base_dir: &Path) -> Result<Vec<String>> {
28    if !base_dir.exists() {
29        return Ok(vec![]);
30    }
31    let mut names = vec![];
32    for entry in fs::read_dir(base_dir)? {
33        let entry = entry?;
34        if entry.file_type()?.is_dir() {
35            if let Some(name) = entry.file_name().to_str() {
36                names.push(name.to_string());
37            }
38        }
39    }
40    names.sort();
41    Ok(names)
42}

44pub fn list_runs(experiment_dir: &Path) -> Result<Vec<String>> {
45    if !experiment_dir.exists() {
46        return Ok(vec![]);
47    }
48    let mut names = vec![];
49    for entry in fs::read_dir(experiment_dir)? {
50        let entry = entry?;
51        if entry.file_type()?.is_dir() {
52            if let Some(name) = entry.file_name().to_str() {
53                // Avoid "artifacts" if we are inside a run, but here we are at experiment level.
54                // However, an experiment folder contains runs (directories).
55                // We should probably filter for directories that contain a run.yaml or metrics.parquet
56                // But for now, just listing all dirs except maybe some reserved ones.
57                if name != "artifacts" {
58                    names.push(name.to_string());
59                }
60            }
61        }
62    }
63    names.sort_by(|a, b| b.cmp(a)); // newest first
64    Ok(names)
65}

67pub fn list_artifacts(run_dir: &Path) -> Result<Vec<ArtifactInfo>> {
68    let mut files = vec![];
69
70    // 1. List default artifacts from run_dir root
71    if run_dir.exists() {
72        for entry in fs::read_dir(run_dir)? {
73            let entry = entry?;
74            let path = entry.path();
75            if path.is_file() {
76                let name = path.file_name().and_then(|n| n.to_str()).unwrap_or("");
77                // Include specific default files
78                if name == "metrics.parquet"
79                    || name == "config.yaml"
80                    || name == "run.yaml"
81                    || name == "run.log"
82                    || name == "console.log"
83                {
84                    let size = path.metadata()?.len();
85                    let ext = path
86                        .extension()
87                        .and_then(|e| e.to_str())
88                        .unwrap_or("")
89                        .to_lowercase();
90                    files.push(ArtifactInfo {
91                        path: name.to_string(),
92                        name: name.to_string(),
93                        size,
94                        ext,
95                        is_default: true,
96                    });
97                }
98            }
99        }
100    }
101
102    // 2. List user artifacts from artifacts/ subdir
103    let artifacts_dir = run_dir.join("artifacts");
104    if artifacts_dir.exists() {
105        collect_files(&artifacts_dir, &artifacts_dir, &mut files)?;
106    }
107
108    Ok(files)
109}

111fn collect_files(root: &Path, dir: &Path, out: &mut Vec<ArtifactInfo>) -> Result<()> {
112    for entry in fs::read_dir(dir)? {
113        let entry = entry?;
114        let path = entry.path();
115        if path.is_dir() {
116            collect_files(root, &path, out)?;
117        } else {
118            let rel = path.strip_prefix(root).unwrap_or(&path);
119            let size = path.metadata()?.len();
120            let ext = path
121                .extension()
122                .and_then(|e| e.to_str())
123                .unwrap_or("")
124                .to_lowercase();
125            out.push(ArtifactInfo {
126                path: rel.to_string_lossy().to_string(),
127                name: path
128                    .file_name()
129                    .and_then(|n| n.to_str())
130                    .unwrap_or("")
131                    .to_string(),
132                size,
133                ext,
134                is_default: false,
135            });
136        }
137    }
138    Ok(())
139}

179pub fn load_run_metadata(run_dir: &Path) -> Result<RunMetadata> {
180    let path = run_dir.join("run.yaml");
181    if !path.exists() {
182        // Construct a minimal metadata from directory name
183        let name = run_dir
184            .file_name()
185            .and_then(|n| n.to_str())
186            .unwrap_or("unknown")
187            .to_string();
188        let exp = run_dir
189            .parent()
190            .and_then(|p| p.file_name())
191            .and_then(|n| n.to_str())
192            .unwrap_or("unknown")
193            .to_string();
194        return Ok(RunMetadata {
195            name,
196            experiment: exp,
197            status: RunStatus::Crashed,
198            started_at: Utc::now(),
199            ..Default::default()
200        });
201    }
202    let content = std::fs::read_to_string(&path)?;
203    Ok(serde_yaml::from_str(&content)?)
204}

216/// Append metric rows to a Parquet file.
217/// Strategy: read existing → concat → write back.
218/// This is called infrequently (batched), so O(n) is acceptable.
219/// For very large files, a future optimization is columnar append via IPC.
220pub fn append_metrics(path: &Path, rows: &[MetricRow]) -> Result<()> {
221    if rows.is_empty() {
222        return Ok(());
223    }
224
225    // Build new batch from rows
226    let new_batch = rows_to_record_batch(rows)?;
227
228    // If file exists, read and concat
229    let final_batch = if path.exists() {
230        let existing = read_parquet(path)?;
231        concat_batches(&existing, &new_batch)?
232    } else {
233        new_batch
234    };
235
236    write_parquet(path, &final_batch)?;
237    Ok(())
238}

240/// Read all metrics from a Parquet file as a list of row maps.
241pub fn read_metrics(path: &Path) -> Result<Vec<HashMap<String, serde_json::Value>>> {
242    if !path.exists() {
243        return Ok(vec![]);
244    }
245    let batch = read_parquet(path)?;
246    record_batch_to_rows(&batch)
247}

249/// Read metrics since a given step (for live streaming).
250pub fn read_metrics_since(
251    path: &Path,
252    since_step: Option<u64>,
253) -> Result<Vec<HashMap<String, serde_json::Value>>> {
254    let all = read_metrics(path)?;
255    if let Some(since) = since_step {
256        Ok(all
257            .into_iter()
258            .filter(|row| {
259                row.get("step")
260                    .and_then(|v| v.as_u64())
261                    .map(|s| s > since)
262                    .unwrap_or(true)
263            })
264            .collect())
265    } else {
266        Ok(all)
267    }
268}

270/// Read the latest row of a parquet file and return only numeric (scalar) columns as f64.
271/// Non-numeric columns (step, timestamp, strings) are silently skipped.
272pub fn read_latest_scalar_metrics(path: &Path) -> Result<HashMap<String, f64>> {
273    if !path.exists() {
274        return Ok(HashMap::new());
275    }
276    let rows = read_metrics(path)?;
277    let Some(last) = rows.into_iter().last() else {
278        return Ok(HashMap::new());
279    };
280    let scalars = last
281        .into_iter()
282        .filter(|(k, _)| k != "step" && k != "timestamp")
283        .filter_map(|(k, v)| v.as_f64().map(|f| (k, f)))
284        .collect();
285    Ok(scalars)
286}

288fn read_parquet(path: &Path) -> Result<RecordBatch> {
289    let file = fs::File::open(path)?;
290    let builder = ParquetRecordBatchReaderBuilder::try_new(file)?;
291    let mut reader = builder.build()?;
292    let mut batches = vec![];
293    for batch in &mut reader {
294        batches.push(batch?);
295    }
296    if batches.is_empty() {
297        // Return empty batch with default schema
298        let schema = Arc::new(Schema::new(vec![
299            Field::new("step", DataType::Int64, true),
300            Field::new(
301                "timestamp",
302                DataType::Timestamp(TimeUnit::Microsecond, Some("UTC".into())),
303                false,
304            ),
305        ]));
306        return Ok(RecordBatch::new_empty(schema));
307    }
308    if batches.len() == 1 {
309        return Ok(batches.remove(0));
310    }
311    // Concat multiple batches
312    let schema = batches[0].schema();
313    Ok(arrow::compute::concat_batches(&schema, &batches)?)
314}

316fn write_parquet(path: &Path, batch: &RecordBatch) -> Result<()> {
317    let file = fs::File::create(path)?;
318    let props = WriterProperties::builder()
319        .set_compression(parquet::basic::Compression::SNAPPY)
320        .build();
321    let mut writer = ArrowWriter::try_new(file, batch.schema(), Some(props))?;
322    writer.write(batch)?;
323    writer.close()?;
324    Ok(())
325}

327fn concat_batches(existing: &RecordBatch, new: &RecordBatch) -> Result<RecordBatch> {
328    // Merge schemas: new batch may have columns not in existing (diagonal concat)
329    let merged_schema = merge_schemas(existing.schema_ref(), new.schema_ref());
330    let merged_schema = Arc::new(merged_schema);
331
332    let existing_aligned = align_batch(existing, &merged_schema)?;
333    let new_aligned = align_batch(new, &merged_schema)?;
334
335    Ok(arrow::compute::concat_batches(
336        &merged_schema,
337        &[existing_aligned, new_aligned],
338    )?)
339}

341fn merge_schemas(a: &Schema, b: &Schema) -> Schema {
342    let mut fields: Vec<Field> = a.fields().iter().map(|f| f.as_ref().clone()).collect();
343    for field in b.fields() {
344        if a.field_with_name(field.name()).is_err() {
345            fields.push(field.as_ref().clone());
346        }
347    }
348    Schema::new(fields)
349}

351fn align_batch(batch: &RecordBatch, target_schema: &Schema) -> Result<RecordBatch> {
352    let n = batch.num_rows();
353    let mut columns: Vec<ArrayRef> = vec![];
354
355    for field in target_schema.fields() {
356        if let Some(col) = batch.column_by_name(field.name()) {
357            columns.push(col.clone());
358        } else {
359            // Fill missing column with nulls
360            let null_array: ArrayRef = match field.data_type() {
361                DataType::Float64 => Arc::new(Float64Array::from(vec![None::<f64>; n])),
362                DataType::Int64 => Arc::new(Int64Array::from(vec![None::<i64>; n])),
363                DataType::Timestamp(TimeUnit::Microsecond, _) => Arc::new(
364                    TimestampMicrosecondArray::from(vec![None::<i64>; n])
365                        .with_timezone_opt(Some("UTC".to_string())),
366                ),
367                _ => Arc::new(StringArray::from(vec![None::<&str>; n])),
368            };
369            columns.push(null_array);
370        }
371    }
372
373    Ok(RecordBatch::try_new(
374        Arc::new(target_schema.clone()),
375        columns,
376    )?)
377}

379fn rows_to_record_batch(rows: &[MetricRow]) -> Result<RecordBatch> {
380    if rows.is_empty() {
381        let schema = Arc::new(Schema::new(vec![
382            Field::new("step", DataType::Int64, true),
383            Field::new(
384                "timestamp",
385                DataType::Timestamp(TimeUnit::Microsecond, Some("UTC".into())),
386                false,
387            ),
388        ]));
389        return Ok(RecordBatch::new_empty(schema));
390    }
391
392    // Collect all unique metric keys across all rows
393    let mut all_keys: Vec<String> = vec![];
394    for row in rows {
395        for key in row.values.keys() {
396            if !all_keys.contains(key) {
397                all_keys.push(key.clone());
398            }
399        }
400    }
401
402    let _n = rows.len();
403
404    // Build columns
405    let mut fields = vec![
406        Field::new("step", DataType::Int64, true),
407        Field::new(
408            "timestamp",
409            DataType::Timestamp(TimeUnit::Microsecond, Some("UTC".into())),
410            false,
411        ),
412    ];
413    let mut arrays: Vec<ArrayRef> = vec![];
414
415    // step column
416    let steps: Vec<Option<i64>> = rows.iter().map(|r| r.step.map(|s| s as i64)).collect();
417    arrays.push(Arc::new(Int64Array::from(steps)));
418
419    // timestamp column (microseconds since epoch UTC)
420    let timestamps: Vec<Option<i64>> = rows
421        .iter()
422        .map(|r| Some(r.timestamp.timestamp_micros()))
423        .collect();
424    arrays.push(Arc::new(
425        TimestampMicrosecondArray::from(timestamps).with_timezone_opt(Some("UTC".to_string())),
426    ));
427
428    // metric value columns
429    for key in &all_keys {
430        // Determine type from first non-null value
431        let first_val = rows.iter().find_map(|r| r.values.get(key));
432        match first_val {
433            Some(MetricValue::Float(_)) | Some(MetricValue::Int(_)) => {
434                // Store as Float64 for simplicity
435                let vals: Vec<Option<f64>> = rows
436                    .iter()
437                    .map(|r| match r.values.get(key) {
438                        Some(MetricValue::Float(f)) => Some(*f),
439                        Some(MetricValue::Int(i)) => Some(*i as f64),
440                        _ => None,
441                    })
442                    .collect();
443                fields.push(Field::new(key, DataType::Float64, true));
444                arrays.push(Arc::new(Float64Array::from(vals)));
445            }
446            _ => {
447                // Store as Utf8
448                let vals: Vec<Option<String>> = rows
449                    .iter()
450                    .map(|r| match r.values.get(key) {
451                        Some(MetricValue::Text(s)) => Some(s.clone()),
452                        Some(MetricValue::Bool(b)) => Some(b.to_string()),
453                        Some(MetricValue::Float(f)) => Some(f.to_string()),
454                        Some(MetricValue::Int(i)) => Some(i.to_string()),
455                        None => None,
456                    })
457                    .collect();
458                fields.push(Field::new(key, DataType::Utf8, true));
459                arrays.push(Arc::new(StringArray::from(vals)));
460            }
461        }
462    }
463
464    let schema = Arc::new(Schema::new(fields));
465    Ok(RecordBatch::try_new(schema, arrays)?)
466}

468fn record_batch_to_rows(batch: &RecordBatch) -> Result<Vec<HashMap<String, serde_json::Value>>> {
469    let schema = batch.schema();
470    let n = batch.num_rows();
471    let mut rows = vec![HashMap::new(); n];
472
473    for (col_idx, field) in schema.fields().iter().enumerate() {
474        let col = batch.column(col_idx);
475        let name = field.name().clone();
476
477        for (row_idx, row) in rows.iter_mut().enumerate().take(n) {
478            use arrow::array::Array;
479            if col.is_null(row_idx) {
480                row.insert(name.clone(), serde_json::Value::Null);
481                continue;
482            }
483            let val = match field.data_type() {
484                DataType::Float64 => {
485                    let arr = col.as_any().downcast_ref::<Float64Array>().unwrap();
486                    let f = arr.value(row_idx);
487                    if f.is_nan() || f.is_infinite() {
488                        serde_json::Value::Null
489                    } else {
490                        serde_json::json!(f)
491                    }
492                }
493                DataType::Int64 => {
494                    let arr = col.as_any().downcast_ref::<Int64Array>().unwrap();
495                    serde_json::json!(arr.value(row_idx))
496                }
497                DataType::Timestamp(TimeUnit::Microsecond, _) => {
498                    let arr = col
499                        .as_any()
500                        .downcast_ref::<TimestampMicrosecondArray>()
501                        .unwrap();
502                    let micros = arr.value(row_idx);
503                    let dt = DateTime::<Utc>::from_timestamp_micros(micros).unwrap_or_default();
504                    serde_json::json!(dt.to_rfc3339())
505                }
506                DataType::Utf8 => {
507                    let arr = col.as_any().downcast_ref::<StringArray>().unwrap();
508                    serde_json::json!(arr.value(row_idx))
509                }
510                _ => serde_json::Value::Null,
511            };
512            row.insert(name.clone(), val);
513        }
514    }
515
516    Ok(rows)
