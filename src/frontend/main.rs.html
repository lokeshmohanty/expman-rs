19fn format_date(iso: &str) -> String {
20    if let Ok(dt) = DateTime::parse_from_rfc3339(iso) {
21        let local = dt.with_timezone(&Local);
22        local.format("%H:%M, %d %b, %Y").to_string()
23    } else {
24        iso.to_string()
25    }
26}

47async fn fetch_experiments() -> Result<Vec<Experiment>, String> {

75async fn update_experiment_metadata(
76    exp_id: String,
77    display_name: Option<String>,
78    description: Option<String>,
79    tags: Option<Vec<String>>,
80) -> Result<(), String> {
81    let payload = serde_json::json!({
82        "display_name": display_name,
83        "description": description,
84        "tags": tags,
85    });
86    let resp = gloo_net::http::Request::patch(&format!("/api/experiments/{}/metadata", exp_id))
87        .json(&payload)
88        .map_err(|e| e.to_string())?
89        .send()
90        .await
91        .map_err(|e| e.to_string())?;
92
93    if !resp.ok() {
94        return Err(format!("Error updating metadata: {}", resp.status()));
95    }
96    Ok(())
97}

346
347    let (selected_runs, set_selected_runs) = signal(std::collections::HashSet::<String>::new());
348    let (active_tab, set_active_tab) = signal("metrics".to_string());
349
350    // Experiment Edit
351    let (show_edit, set_show_edit) = signal(false);
352    let (edit_name, set_edit_name) = signal("".to_string());
353    let (edit_desc, set_edit_desc) = signal("".to_string());
354    let (edit_tags, set_edit_tags) = signal("".to_string());
355
356    // Run Edit
357    let (show_run_edit, set_show_run_edit) = signal(false);
358    let (edit_run_id, set_edit_run_id) = signal("".to_string());
359    let (edit_run_name, set_edit_run_name) = signal("".to_string());
360    let (edit_run_desc, set_edit_run_desc) = signal("".to_string());

363        set_selected_runs.update(|set| {
364            if set.contains(&name) {
365                set.remove(&name);
366            } else {
367                set.insert(name);
368            }
369        });

735fn LineChart(
736    #[allow(unused_variables)] exp_id: String,
737    selected_runs: std::collections::HashSet<String>,
738) -> impl IntoView {
739    let div_ref = NodeRef::<leptos::html::Div>::new();
740
741    Effect::new(move |_| {
742        if let Some(div) = div_ref.get() {
743            let mut p = Plot::new();
744            let layout = Layout::new()
745                .margin(Margin::new().left(50).right(50).top(30).bottom(50))
746                .show_legend(true)
747                .paper_background_color("rgba(0,0,0,0)")
748                .plot_background_color("rgba(0,0,0,0)")
749                .font(plotly::common::Font::new().color("#94a3b8"))
750                .x_axis(
751                    Axis::new()
752                        .title(Title::from("Step"))
753                        .show_grid(true)
754                        .grid_color("#1e293b"),
755                )
756                .y_axis(
757                    Axis::new()
758                        .title(Title::from("Value"))
759                        .show_grid(true)
760                        .grid_color("#1e293b"),
761                );
762
763            p.set_layout(layout);
764
765            // Mock data for now - in real app this would fetch from backend/SSE
766            for run_id in selected_runs.iter() {
767                let x: Vec<f64> = (0..20).map(|i| i as f64).collect();
768                let y: Vec<f64> = (0..20)
769                    .map(|i| {
770                        let base = (i as f64).sin();
771                        base + (run_id.len() as f64 % 10.0) / 10.0
772                    })
773                    .collect();
774
775                let trace = Scatter::new(x, y)
776                    .name(run_id.as_str())
777                    .mode(plotly::common::Mode::LinesMarkers);
778                p.add_trace(trace);
779            }
780
781            // Serialize to JSON string using plotly's to_json
782            let json_str = p.to_json();
783
784            // Parse JSON string to JS object
785            if let Ok(js_value) = js_sys::JSON::parse(&json_str) {
786                let data =
787                    js_sys::Reflect::get(&js_value, &"data".into()).unwrap_or(JsValue::UNDEFINED);
788                let layout =
789                    js_sys::Reflect::get(&js_value, &"layout".into()).unwrap_or(JsValue::UNDEFINED);
790                let config =
791                    js_sys::Reflect::get(&js_value, &"config".into()).unwrap_or(JsValue::UNDEFINED);
792
793                let div_element: &web_sys::HtmlElement = &div;
794                new_plot(&div_element.into(), &data, &layout, &config);
795            } else {
796                leptos::logging::error!("Failed to parse Plotly JSON");
797            }
798        }
799    });
800
801    view! {
802        <div class="w-full h-full p-2">
803            <div node_ref=div_ref class="w-full h-full"></div>
804        </div>
805    }

808#[component]
809fn TabularPreview(content: String) -> impl IntoView {
810    // Try to parse as JSON first (backend sends {type: "parquet", data: [...]})
811    if let Ok(json) = serde_json::from_str::<serde_json::Value>(&content) {
812        if json["type"] == "parquet" {
813            if let Some(data) = json["data"].as_array() {
814                if data.is_empty() {
815                    return view! { <div class="p-8 text-slate-500 italic">"No data available in this parquet file."</div> }.into_any();
816                }
817
818                let headers: Vec<_> = data[0]
819                    .as_object()
820                    .map(|obj| obj.keys().cloned().collect())
821                    .unwrap_or_default();
822
823                return view! {
824                    <div class="overflow-auto max-h-full">
825                        <table class="w-full text-left border-collapse min-w-max">
826                            <thead class="sticky top-0 bg-slate-900 border-b border-slate-800">
827                                <tr>
828                                    {headers.iter().cloned().map(|h| view! {
829                                        <th class="p-3 text-[10px] font-bold text-slate-400 uppercase tracking-wider">{h}</th>
830                                    }).collect_view()}
831                                </tr>
832                            </thead>
833                            <tbody class="divide-y divide-slate-800/50">
834                                {data.iter().map(|row| {
835                                    let fields: Vec<_> = headers.iter().map(|h| row[h].to_string().replace("\"", "")).collect();
836                                    view! {
837                                        <tr class="hover:bg-slate-800/30 transition-colors">
838                                            {fields.into_iter().map(|f| view! {
839                                                <td class="p-3 text-slate-300 font-mono">{f}</td>
840                                            }).collect_view()}
841                                        </tr>
842                                    }
843                                }).collect_view()}
844                            </tbody>
845                        </table>
846                    </div>
847                }.into_any();
848            }
849        }
850    }
851
852    // Default: check if it looks like CSV
853    if content.contains(',') && content.lines().count() > 1 {
854        let lines: Vec<&str> = content.lines().collect();
855        let headers: Vec<String> = lines[0].split(',').map(|s| s.trim().to_string()).collect();
856        let rows: Vec<Vec<String>> = lines[1..]
857            .iter()
858            .map(|line| line.split(',').map(|s| s.trim().to_string()).collect())
859            .collect();
860
861        return view! {
862            <div class="overflow-auto max-h-full">
863                <table class="w-full text-left border-collapse min-w-max">
864                    <thead class="sticky top-0 bg-slate-900 border-b border-slate-800">
865                        <tr>
866                            {headers.into_iter().map(|h| view! {
867                                <th class="p-3 text-[10px] font-bold text-slate-400 uppercase tracking-wider">{h}</th>
868                            }).collect_view()}
869                        </tr>
870                    </thead>
871                    <tbody class="divide-y divide-slate-800/50">
872                        {rows.into_iter().map(|row| {
873                            view! {
874                                <tr class="hover:bg-slate-800/30 transition-colors">
875                                    {row.into_iter().map(|f| view! {
876                                        <td class="p-3 text-slate-300 font-mono">{f}</td>
877                                    }).collect_view()}
878                                </tr>
879                            }
880                        }).collect_view()}
881                    </tbody>
882                </table>
883            </div>
884        }.into_any();
885    }
886
887    // Fallback to text
888    view! { <div class="whitespace-pre p-4">{content}</div> }.into_any()
889}

899async fn fetch_artifacts(exp_id: String, run_id: String) -> Result<Vec<Artifact>, String> {
900    let resp = gloo_net::http::Request::get(&format!(
901        "/api/experiments/{}/runs/{}/artifacts",
902        exp_id, run_id
903    ))
904    .send()
905    .await
906    .map_err(|e| e.to_string())?;
907
908    if !resp.ok() {
909        return Err(format!("Error fetching artifacts: {}", resp.status()));
910    }
911
912    let text = resp.text().await.map_err(|e| e.to_string())?;
913    serde_json::from_str(&text).map_err(|e| e.to_string())
914}

1061    });
1062

1160#[component]
1161fn RunsTableView(exp_id: String) -> impl IntoView {
1162    let runs = LocalResource::new(move || {
1163        let id = exp_id.clone();
1164        async move { fetch_runs(id).await }
1165    });
1166
1167    // Which metric columns are currently visible (None = all visible)
1168    let (selected_metrics, set_selected_metrics) =
1169        signal(std::collections::HashSet::<String>::new());
1170    let (metrics_initialized, set_metrics_initialized) = signal(false);
1171
1172    view! {
1173        <div class="flex-grow p-6 overflow-auto space-y-4">
1174             <Suspense fallback=|| view! { <div class="p-4 text-center text-slate-500">"Loading runs..."</div> }>
1175                {move || Suspend::new(async move {
1176                    let run_list = runs.await.unwrap_or_default();
1177                    if run_list.is_empty() {
1178                        return view! { <div class="p-12 text-center text-slate-500">"No runs found for this experiment."</div> }.into_any();
1179                    }
1180
1181                    // Collect all unique metric keys (sorted)
1182                    let all_metric_keys: Vec<String> = {
1183                        let mut keys = std::collections::BTreeSet::new();
1184                        for run in &run_list {
1185                            if let Some(metrics) = &run.metrics {
1186                                for key in metrics.keys() {
1187                                    keys.insert(key.clone());
1188                                }
1189                            }
1190                        }
1191                        keys.into_iter().collect()
1192                    };
1193
1194                    // Initialize selected_metrics to all keys on first load
1195                    if !metrics_initialized.get() && !all_metric_keys.is_empty() {
1196                        set_selected_metrics.set(all_metric_keys.iter().cloned().collect());
1197                        set_metrics_initialized.set(true);
1198                    }
1199
1200                    let keys_for_filter = all_metric_keys.clone();
1201                    let keys_for_table = all_metric_keys.clone();
1202
1203                    view! {
1204                        // ── Metric filter chips ──────────────────────────────────
1205                        {if !keys_for_filter.is_empty() {
1206                            view! {
1207                                <div class="flex flex-wrap items-center gap-2">
1208                                    <span class="text-xs font-semibold text-slate-500 uppercase tracking-wider mr-1">"Metrics:"</span>
1209                                    {keys_for_filter.into_iter().map(|key| {
1210                                        let k1 = key.clone();
1211                                        let k2 = key.clone();
1212                                        let is_on = Signal::derive(move || selected_metrics.with(|s| s.contains(&k1)));
1213                                        view! {
1214                                            <button
1215                                                on:click=move |_| {
1216                                                    let k = k2.clone();
1217                                                    set_selected_metrics.update(|s| {
1218                                                        if s.contains(&k) { s.remove(&k); } else { s.insert(k); }
1219                                                    });
1220                                                }
1221                                                class=move || format!(
1222                                                    "px-3 py-1 rounded-full text-xs font-medium border transition-all duration-150 {}",
1223                                                    if is_on.get() {
1224                                                        "bg-blue-600/20 border-blue-500/50 text-blue-300 hover:bg-blue-600/30"
1225                                                    } else {
1226                                                        "bg-slate-800 border-slate-700 text-slate-500 hover:border-slate-600 hover:text-slate-400"
1227                                                    }
1228                                                )
1229                                            >
1230                                                {key}
1231                                            </button>
1232                                        }
1233                                    }).collect_view()}
1234                                </div>
1235                            }.into_any()
1236                        } else {
1237                            view! { <div></div> }.into_any()
1238                        }}
1239
1240                        // ── Runs table ───────────────────────────────────────────
1241                        <div class="bg-slate-900 border border-slate-800 rounded-xl overflow-hidden">
1242                            <table class="w-full text-left border-collapse">
1243                                <thead class="bg-slate-950 text-xs uppercase text-slate-500 font-semibold sticky top-0">
1244                                    <tr>
1245                                        <th class="p-4 border-b border-slate-800">"Run ID"</th>
1246                                        <th class="p-4 border-b border-slate-800">"Status"</th>
1247                                        {keys_for_table.iter().filter(|k| selected_metrics.with(|s| s.contains(*k))).map(|k| view! {
1248                                            <th class="p-4 border-b border-slate-800 text-blue-400">{k.clone()}</th>
1249                                        }).collect_view()}
1250                                        <th class="p-4 border-b border-slate-800">"Duration"</th>
1251                                        <th class="p-4 border-b border-slate-800">"Started"</th>
1252                                        <th class="p-4 border-b border-slate-800">"Description"</th>
1253                                    </tr>
1254                                </thead>
1255                                <tbody class="divide-y divide-slate-800/50 text-sm text-slate-300">
1256                                    {run_list.into_iter().map(|run| {
1257                                        let duration = run.duration_secs.map(|d| format!("{:.1}s", d)).unwrap_or("-".to_string());
1258                                        let (status_color, status_bg, status_border) = match run.status.as_str() {
1259                                            "RUNNING"   => ("text-blue-400",   "bg-blue-500",   "border-blue-500"),
1260                                            "COMPLETED" => ("text-emerald-400", "bg-emerald-500", "border-emerald-500"),
1261                                            "FAILED"    => ("text-red-400",     "bg-red-500",     "border-red-500"),
1262                                            _           => ("text-slate-400",   "bg-slate-600",   "border-slate-500"),
1263                                        };
1264                                        let dot_class = if run.status == "RUNNING" { "animate-pulse" } else { "" };
1265                                        let run_metrics = run.metrics.clone().unwrap_or_default();
1266                                        let metric_cols: Vec<String> = keys_for_table.iter()
1267                                            .filter(|k| selected_metrics.with(|s| s.contains(*k)))
1268                                            .cloned()
1269                                            .collect();
1270
1271                                        view! {
1272                                            <tr class="hover:bg-slate-800/30 transition-colors group">
1273                                                <td class="p-4 font-mono text-white flex items-center space-x-2">
1274                                                    <div class=format!("w-2 h-2 rounded-full {} {}", status_bg, dot_class)></div>
1275                                                    <span>{run.name}</span>
1276                                                </td>
1277                                                <td class="p-4">
1278                                                    <span class=format!("px-2 py-1 rounded text-xs font-medium bg-opacity-10 border border-opacity-20 {} {} {}", status_bg, status_color, status_border)>
1279                                                        {run.status}
1280                                                    </span>
1281                                                </td>
1282                                                {metric_cols.into_iter().map(|k| {
1283                                                    let val = run_metrics.get(&k)
1284                                                        .map(|f| format!("{:.4}", f))
1285                                                        .unwrap_or_else(|| "-".to_string());
1286                                                    view! { <td class="p-4 font-mono text-slate-400">{val}</td> }
1287                                                }).collect_view()}
1288                                                <td class="p-4 font-mono text-slate-400">{duration}</td>
1289                                                <td class="p-4 text-slate-400 whitespace-nowrap">{format_date(&run.started_at)}</td>
1290                                                <td class="p-4 text-slate-500 truncate max-w-xs group-hover:text-slate-300 transition-colors">{run.description.unwrap_or_default()}</td>
1291                                            </tr>
1292                                        }
1293                                    }).collect_view()}
1294                                </tbody>
1295                            </table>
1296                        </div>
1297                    }.into_any()
1298                })}
1299             </Suspense>
1300        </div>
1301    }
