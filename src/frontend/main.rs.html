20fn format_date(iso: &str) -> String {
21    if let Ok(dt) = DateTime::parse_from_rfc3339(iso) {
22        let local = dt.with_timezone(&Local);
23        local.format("%H:%M, %d %b, %Y").to_string()
24    } else {
25        iso.to_string()
26    }
27}

78async fn update_experiment_metadata(
79    exp_id: String,
80    display_name: Option<String>,
81    description: Option<String>,
82    tags: Option<Vec<String>>,
83) -> Result<(), String> {
84    let payload = serde_json::json!({
85        "display_name": display_name,
86        "description": description,
87        "tags": tags,
88    });
89    let resp = gloo_net::http::Request::patch(&format!("/api/experiments/{}/metadata", exp_id))
90        .json(&payload)
91        .map_err(|e| e.to_string())?
92        .send()
93        .await
94        .map_err(|e| e.to_string())?;
95
96    if !resp.ok() {
97        return Err(format!("Error updating metadata: {}", resp.status()));
98    }
99    Ok(())
100}
